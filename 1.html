(tno char(3) not null primary key, tname varchar(8), tsex char(2) check(tsex in('男','女')), tbirthday date, tdept char(16)) go

alter table teacher add default '信息系' for tdept go

--3插入两位老师的信息，如下

--102，张金，男，软件系

--106，肖小，女

insert into teacher(tno,tname,tsex,tdept) values('102','张金','男','软件系'), ('106','肖小','女',null) go

--4.修改'106'编号老师的出生日期为1996.6.9

update teacher set tbirthday='1996-06-09' where tno='106' go

--5.删除肖小老师的数据

delete from teacher where tname='肖小' go

--a根据上述三表，请查询出每位老师的授课门数，显示教师号以及授课门数

select tno,count(*) as 课程门数 from 授课表 group by cno,tno go

--b根据上述三表，请查出'李新'老师的授课门数，显示教师名授课门数

select count(*) as 课程门数,tname from 教师表 join 授课表 on 教师表.tno=授课表.tno where tname='李新' group by cno,tname go

--c根据上述三表，请查出上'数学'课程的老师的信息，显示课程名称，教师号，所在系。

select cname,教师表.tno,tdept from 教师表 join 授课表 on 教师表.tno=授课表.tno join 课程名称表 on 授课表.cno=课程名称表.cno where cname='数学' go

--d根据上诉三表，查出只授课一门课程的老师信息，显示教师名，所在系，授课门数

select count(*) as 课程门数,tname,tdept from 教师表 join 授课表 on 教师表.tno=授课表.tno group by cno,tname,tdept having count(*)=1 go

--e）根据上述三表，查询所有老师的基本信息，显示教师名、所在系、年龄，按年龄降序排。

select tname,tdept,YEAR (GETDATE())-YEAR (tbirthday) as 年龄 from 教师表 order by 年龄 desc go

--f）根据上述三表，查询出课程的授课老师只有1人的信息，显示课程号、课程名、授课老师人数。

select 授课表.cno,cname,count (tno) as 授课老师人数 from 授课表 join 课程名称表 on 授课表.cno=课程名称表.cno group by 授课表.cno, cname, tno having count (tno)=1 go

--g）根据上述三表，创建一个存储过程prc_getTeacherInfo，查询每位教师的上课门数；

--最终希望显示每位教师的教工号、姓名、上课门数。

create procedure prc_getTeacherInfo as select 教师表.tno,tname,count(*) as 上课门数 from 教师表 join 授课表 on 教师表.tno=授课表.tno group by 教师表.tno, tname go exec prc_getTeacherInfo go
